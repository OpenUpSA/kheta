#### **Using createComponentWithLink**

You can use `createComponentWithLink` if you don't want to pass in `Link` all the time.

```jsx
import React from 'react';
import { Link as link } from 'gatsby';
import { createComponentWithLink } from './src/components/GradientButton';

const ButtonWithLink = createComponentWithLink({ link });

return () => (
  <div>
    <ButtonWithLink clickAction="https://google.com/example-1" /> {/* Opens new tab */}
    <ButtonWithLink clickAction="/example-3" /> {/* Does not open new tab */}
    <ButtonWithLink clickAction="//www.google.com" /> {/* Opens new tab */}
    <ButtonWithLink clickAction="folder/example-3" /> {/* Does not open new tab */}
  </div>
)
```
